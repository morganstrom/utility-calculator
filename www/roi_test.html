Monte-Carlo Simulation of utility and regression error

Author: Morgan Strm 2014

We start by defining the input variables and calculating important parameters.

library(MASS)
library(reshape2)
library(ggplot2)

#Input variables
npositions <- 20 #Number of positions
ncandidates <- 100 #Number of candidates applying 
years <- 3 #Number of years each employee is expected to stay
salary <- 600000 #Salary for the role
r <- 0.51 #Predictive validity of assessment
C <- 25000 #Cost of each assessment

#Calculation of important parameters
SDy <- salary * 0.4 #standard deviation of utility, based on general rule from Schmidt, Hunter & Pearlman (1982) 
p <- npositions/ncandidates #proportion of candidates we wish to hire (selection ratio)
cut <- qnorm(1-p, lower.tail=TRUE) #Quantile function of the normal curve, giving the z-score corresponding to given selection ratio
phi <- dnorm(cut) #ordinate (density) in normal distribution at cut-off
zs <- phi / p #mean of test score in the selected group
sigma <- matrix(c(1, r, r, 1), 2, 2)  #Correlation matrix for bivariate normal distribution

We can then calculate the estimated utility U and the standard error of the estimate e, as done in the visualization.

#Analytical estimation of utility and standard error of the estimate
U <- npositions * years * r * SDy * zs - ncandidates * C
e <- sqrt(1-r^2)*SDy*years

This will later be compared to the Monte Carlo estimates of utility and standard error.

Now, we first define the number of iterations for the simulation and create vectors to hold the values. Then we create a loop where a sample from a bivariate normal distribution will be generated in each iteration, from which estimates of utility and standard error will be calculated.

R <- 5000 #Number of iterations
est <- numeric(length=R) #data vector for estimations of utility
err <- numeric(length=R) #data vector for estimations of prediction error

#Simulation loop
for(i in 1:R) {
  #sample from bivariate normal distribution with assumed correlation matrix
  biv_dist <- mvrnorm(n=ncandidates, mu=c(0,0), Sigma=sigma, empirical=FALSE) 
  
  x <- biv_dist[,1] #Predictor variable
  y <- biv_dist[,2] #Outcome variable
  
  #identify observations above cut-off in x
  j <- x > cut 
  
  #Calculate predicted values from the linear model for all observations
  pred <- years * SDy * r * x 
  #Transpose outcome variable for all observations to the correct unit
  true <- years * SDy * y 
  
  #Calculate estimation of total utility (for the hired applicants)
  est[i] <- sum(pred[j]) - ncandidates*C 
  
  #Calculate standard error of the regression
  #s^2 = SSR / (n - K) = sum((true-pred)^2) / (n - K)
  #Hayashi, 2000, p. 19
  n <- length(pred)
  if(n>1) err[i] <- sqrt(sum((true - pred)^2) / (n-1)) 
  else err[i] <- NA #Note that the error cannot be estimated if K is equal to or larger than n
  
}

This procedure allows for investigating the effect of random sampling error on the estimates. Below is a histogram of the utility estimates from the Monte Carlo simulation above.

hist(est, main="Monte Carlo sampling distribution of utility", xlab="Utility")
abline(v=mean(est, na.rm=T), col="blue")
abline(v=U, col="red")
legend(x=U+1.5*sd(est,na.rm=T), y=700, legend=c("Monte Carlo", "Formula"), col=c("blue", "red"), lty=1)

This was a single Monte Carlo simulation using fixed values of all variables. Will the results from the formulas be close to simulated results even with other values for the variables? To investigate, we can make simulations over varying values of the independent variables r, p and N. Well set C, SDy and years as constants.

#Independent variables
p <- 1:9/10 #9 different selection ratios
r <- seq(0, 1, 0.1) #11 different validity coefficients
ncandidates <- c(10, 20, 30, 40, 50, 60) #6 different numbers of applicants

#Constants
C <- 75 #cost per assessment in euros
SDy <- 20000 #standard deviation of utility, based on annual salary of 50 000 euros
years <- 1

Now, the simulation will be more complicated since it will iterate over all the levels of the independent variables. With 100 iterations, this means 500911*6 = 297 000 simulated datasets, which may take a couple of minutes to compute with a modern laptop. Simulations with more iterations give more stable results but take longer time. The data objects to store the simulated values of utility and error will be arrays with 4 dimensions.

#Number of iterations
R <- 500

#Dependent variables - arrays in 4 dimensions
U <- array(dim=c(length(r), length(p), length(ncandidates), R), dimnames=c("r", "p", "n", "R"))
e <- array(dim=c(length(r), length(p), length(ncandidates), R), dimnames=c("r", "p", "n", "R"))

Below is the actual simulation. You will recognize the calculations from above, only with more index variables to keep track of..

for(i in 1:length(r)) { #Loop through the different validation coefficients
  sigma <- matrix(c(1, r[i], r[i], 1), nrow=2, ncol=2)
  
  for(j in 1:length(p)) { #Loop through the different selection ratios
    cut <- qnorm(1-p[j], lower.tail=TRUE)
    
    for(k in 1:length(ncandidates)) { #Loop through different number of candidates
    
     for(l in 1:R) { #Monte Carlo iterations
      
      #sample from bivariate normal distribution with assumed correlation matrix
      biv_dist <- mvrnorm(n=ncandidates[k], mu=c(0,0), Sigma=sigma, empirical=FALSE) 
      
      x <- biv_dist[,1] #Predictor variable
      y <- biv_dist[,2] #Outcome variable
      
      m <- x > cut #identify observations above cut-off in x
      
      pred <- SDy * r[i] * x #Calculate predicted values from the linear model for all observations
      true <- SDy * y #Transpose outcome variable for all observations to the correct unit
      
      
      #Calculations
      U[i,j,k,l] <- sum(pred[m], na.rm=TRUE) - ncandidates[k]*C #Calculate estimation of mean utility (for the hired applicants)
      
      #Calculate standard error of the regression 
      n <- length(pred)
      if(n>1) e[i,j,k,l] <- sqrt(sum((true - pred)^2, na.rm=T) / (n-1)) 
      else e[i,j,k,l] <- NA
     }
     
    }
    
  }
  
}

Now, lets calculate means for all the iterations and create a data frame in the correct format for plotting.

As a comparison, lets calculate the corresponding values from our formulas.

#Utility
U.formula <- array(dim=c(length(r), length(p), length(ncandidates)))
for(i in 1:length(p)) {
  for(j in 1:length(ncandidates)) {
    cut <- qnorm(1-p[i], lower.tail=TRUE) 
    phi <- dnorm(cut)
    U.formula[,i,j] <- ncandidates[j] * r * SDy * phi - ncandidates[j] * C
  }
  
}

U.formula <- melt(U.formula, varnames=c("r", "p", "n"))
U.formula$r <- r
U.formula$p <- rep(p, each=length(r))
U.formula$n <- gl(n=length(ncandidates), k=length(r)*length(p), 
                  labels=c("N = 10", "N = 20", "N = 30", "N = 40", "N = 50", "N = 60"))

#Error
e.formula <- array(dim=c(length(r), length(p), length(ncandidates)))
for(i in 1:length(p)) {
  for(j in 1:length(ncandidates)) {
    e.formula[,i,j] <- sqrt(1-r^2)*SDy
  }
}

e.formula <- melt(e.formula, varnames=c("r", "p", "n"))
e.formula$r <- r
e.formula$p <- rep(p, each=length(r))
e.formula$n <- gl(n=length(ncandidates), k=length(r)*length(p), 
                  labels=c("N = 10", "N = 20", "N = 30", "N = 40", "N = 50", "N = 60"))

Below is a plot comparing the results from the Monte Carlo simulation with the analytical estimates. The results are almost identical. Other insights from this plot are 1. that a selection ratio of 0.5 always gives the highest utility 2. that utilities increase with the number of applicants (and therefore also the number of selectees) 3. that utilities are higher with increasing validity.

compare <- rbind(mean.U, U.formula)
compare$method <- gl(2, length(r)*length(p)*length(ncandidates), labels=c("MC Estimate", "Formula"))

ggplot(compare, aes(p, value, colour=r, group=r)) + geom_line() + facet_grid(method~n) +

The final graph below compares the error estimates from the Monte Carlo simulation to those from the formula. Note that the formula tends to underestimate the error resulting from random sampling slightly when the number of applicants is smaller than 20. For this reason, the results from the visualization should be interpreted with caution with small sample sizes.
